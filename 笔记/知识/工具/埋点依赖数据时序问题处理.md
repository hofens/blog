
1、埋点曝光依赖某个数据

对象增加属性 exposureListeners，存放业务回调；
```kotlin
  
class FamilyDmBean {  
    var family_id: String? = null 
  
    var isInitialized = false  
        set(value) {  
            field = value  
            if (value) {  
                CLog.d("家族埋点", "family_id被赋值 开始回调 ${exposureListeners.size}")  
                exposureListeners.forEach {  
                    it.invoke()  
                }  
            }  
        }  
  
    private var exposureListeners = mutableListOf<() -> Unit?>()  
  
    /**  
     * 如果FamilyDmBean还没初始化，回调Listener就加入集合中，等初始化会遍历回调  
     * FamilyDmBean已经初始化，立即执行回调  
     */  
    fun addExposureCallback(function: () -> Unit?) {  
        if (isInitialized) {  
            function.invoke()  
        } else {  
            exposureListeners.add(function)  
        }  
        CLog.d("家族埋点", "添加监听 $exposureListeners")  
    }  
}

host.familyInfoViewModel.dmBean.addExposureCallback {  
    familyTodoRepo.gangUpInfo?.let { _it->  
        FamilyDmHelper.dmFamilyTodoExposure(host.familyInfoViewModel.dmBean, _it.desc)  
    }  
}
```
